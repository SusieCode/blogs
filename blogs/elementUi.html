    1. 在components文件夹中创建v-table文件
    <template>
  <div class="v-table">
    <el-table
      v-bind="attrs"
      v-on="$listeners"
      ref="elTable">
      <el-table-column
        v-if="showSelect"
        type="selection"
        width="55" />
      <el-table-column
        v-if="showIndex"
        label="序号"
        width="55">
        <template v-slot="{ $index }">
          {{ $index + (pageOption.pageIndex - 1) * pageOption.pageSize + 1 }}
        </template>
      </el-table-column>
      <template v-for="column in columns">
        <el-table-column
          :key="column.prop"
          v-bind="getColumn(column)"
          :label="column.label"
          show-overflow-tooltip>
          <!-- 表头插槽 -->
          <template
            v-if="column.headerSlot"
            #header="slotProps">
            <slot
              :name="column.prop + 'Header'"
              v-bind="slotProps" />
          </template>
          <!-- 列插槽 -->
          <template
            v-if="column.slot"
            v-slot="slotProps">
            <slot
              :name="column.prop"
              v-bind="slotProps" />
          </template>
          <template
            v-else-if="column.formatDate || column.number !== undefined"
            v-slot="{ row }">
            <!-- 日期格式化 -->
            <div v-if="column.formatDate">
              {{ formatDate(row[column.prop], column.formatDate) }}
            </div>
            <!-- 金额千分位格式化 -->
            <div v-else-if="column.number">
              {{ formatNumber(row[column.prop], column.number) }}
            </div>
          </template>
        </el-table-column>
      </template>
    </el-table>
    <div
      v-if="pager.total !== 0"
      class="pagination">
      <el-pagination
        background
        :layout="pageLayout"
        :page-sizes="pageSizes"
        :page-size="pager.pageSize"
        :total="pager.total"
        :current-page.sync="pager.pageIndex"
        @current-change="currentChange"
        @size-change="sizeChange"></el-pagination>
    </div>
  </div>
</template>

<script>
export default {
  props: {
    // 表格列, slot:true 可使用插槽
    columns: Array,
    // 是否显示序号
    showIndex: {
      type: Boolean,
      default: false
    },
    // 是否显示选择列
    showSelect: {
      type: Boolean,
      default: false
    },
    // 支持sync修饰符,可同步更改父组件的数据
    pageOption: {
      type: Object,
      default: () => ({
        total: 0
      })
    },
    pageLayout: {
      type: String,
      default: 'total, prev, pager, next, sizes, jumper'
    },
    pageSizes: {
      type: Array,
      default: () => [5, 10, 20, 50, 100, 200]
    }
  },
  data() {
    return {
      pager: {
        pageIndex: 1,
        pageSize: 10,
        total: 0
      }
    };
  },
  computed: {
    attrs() {
      const attrs = this.$attrs;
      if (!(attrs['header-cell-style'] || attrs['headerCellStyle'])) {
        attrs.headerCellStyle = {
          background: '#f0f4f9',
          color: '#323232',
          fontSize: '14px',
          textAlign: 'center'
        };
      }
      if (!(attrs['cell-style'] || attrs['cellStyle'])) {
        attrs.cellStyle = {
          fontSize: '12px',
          textAlign: 'center'
        };
      }
      if (!attrs['size']) {
        attrs.size = 'medium';
      }
      if (['small', 'mini'].includes(attrs['size'])) {
        attrs.headerCellStyle.fontSize = '12px';
      }
      return attrs;
    }
  },
  watch: {
    pageOption: {
      deep: true,
      immediate: true,
      handler() {
        let { pageIndex, pageSize, total } = this.pageOption;
        this.pager.pageIndex = pageIndex || 1;
        this.pager.pageSize = pageSize || 10;
        this.pager.total = total;
      }
    }
  },
  methods: {
    formatDate(date, format) {
      return this.$common.formatDate(date, format);
    },
    formatNumber(number, fixCount) {
      return this.$common.formatNumber(number, fixCount);
    },
    currentChange(pageIndex) {
      this.pageOptionEmit(pageIndex);
    },

    sizeChange(pageSize) {
      this.pageOptionEmit(undefined, pageSize);
    },

    pageOptionEmit(index, size) {
      if (index) this.pager.pageIndex = index;
      if (size) this.pager.pageSize = size;
      const pageOption = {};
      const { pageIndex, pageSize, total } = this.pager;
      Object.assign(pageOption, this.pager);

      this.$emit('update:pageOption', pageOption);
      // 解决翻到最后一页，在把每页显示条数改成最大的时候，触发的sizeChange会不准确问题
      if (pageIndex > Math.ceil(total / pageSize)) return;
      this.$emit('pageChange');
    },
    // 要删除slot属性,不然v-bind的时候有slot属性,这列会不显示
    getColumn(column) {
      const copyColumn = JSON.parse(JSON.stringify(column));
      delete copyColumn.slot;
      return copyColumn;
    }
  }
};
</script>

    <template>
  <div>
    <el-table
      :data="rsuTableData"
      :span-method="mergeLines"
      :header-cell-style="headerCellStyle"
      :cell-style="cellStyle"
      :row-class-name="tableRowClassName"

          return {
            rowspan: param.row.sameIndex,
            colspan: 1

          return {

            colspan: 0

      if (param.columnIndex === 2) {
        return {
          borderRight: '1px solid  #EBEEF5',
          textAlign: 'center',
          fontSize: '14px'
        };
      }
      return { textAlign: 'center', fontSize: '12px' };
    },
    // 判断决定是否修改样式，如果分组一致则使用同一个样式
    tableRowClassName({ row }) {
      if (row.order == this.currentIndex) {
        return 'hover-bg';
      } else {
        return '';
      }
    },
    // 鼠标移入
    handleCellMouseEnter(row) {
      //标记当前是哪个分组
      this.currentIndex = row.order;
    },
    // 鼠标移出
    handleCellMouseLeave() {
      this.currentIndex = '';
    },
    // 合计
    getRsuSummaries({ columns }) {
      const sums = [];
      columns.forEach((column, i) => {
        if (i === 0) {
          sums[i] = '总计-RSU';
          return;
        }
        if (column.property === 'amount') {
          return (sums[i] = this.ltiInfo?.totalRsu);
        }
        if (column.property === 'rate') {
          return (sums[i] = this.ltiInfo?.rsuRate * 100 + '%');
        }
        if (column.property === 'saveAmount') {
          return (sums[i] = this.ltiInfo?.totalSaveRsu);
        }
        if (column.property === 'cancelAmount') {
          return (sums[i] = this.ltiInfo?.totalCancelRsu);
        }
      });
      return sums;
    }
  }
};
</script>

      roleAuthData: [],
      defaultParams: {
        id: 'id',
        label: 'name',
        children: 'children'
      }
    };
  },
  // 过滤
  watch: {
    filterText(val) {
      this.$refs.tree.filter(val);
    }
  },
  methods: {
    closeDialog() {
      this.$nextTick(() => {
        this.$refs.tree.setCheckedKeys([]);
      });
    },
    async handleConfirm() {
      if (!this.roleForm.name) {
        this.$message.warning('角色名不能为空');
        return;
      }
      if (!this.roleForm.remark) {
        this.$message.warning('备注不能为空');
        return;
      }
      if (this.roleOperateType === roleOperateEnum.add) {
        // 添加角色
        const data = {
          name: this.roleForm.name,
          remark: this.roleForm.remark,
          permissions: this.checkedNodeArr
        };
        await RoleSvc.createRole(data);
        this.getRoleList();
        this.$message.success('操作成功');
        this.roleDialogVisible = false;
      } else if (this.roleOperateType === roleOperateEnum.amend) {
        // 修改角色
        const params = {
          name: this.roleForm.name,
          remark: this.roleForm.remark,
          permissions: this.checkedNodeArr
        };
        await RoleSvc.updateRole(params, this.activeRoleObj.id);
        this.getRoleList();
        this.$message.success('操作成功');
        this.roleDialogVisible = false;
      }
    },
    handleCancel() {
      this.roleDialogVisible = false;
      this.roleForm.name = this.roleForm.remark = '';
    },
    handleCheckChange() {
      // 获取所有选中的节点(叶子节点、半选节点)
      let checkedNode = this.$refs.tree.getCheckedNodes(false, true);
      const arr = [];
      checkedNode.forEach(item => {
        arr.push(item.id);
      });
      this.checkedNodeArr = arr.filter(node => node > 0);
    },
    // 过滤功能项
    filterRole(value, data) {
      if (!value) return true;
      return data.name.toLowerCase().indexOf(value.toLowerCase()) !== -1;
    }
  }
};
</script>
