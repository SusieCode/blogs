    # 一、获取 PDF 文档流直接打印并下载

    • 项目需求：
      1. 前端点击按钮直接调起打印对应 pdf 文档；
      2. 点击按钮打印 pdf 文档到本地
        (使用file-saver.js (opens new window)是在客户端保存文件的解决方案，非常适合在客户端生成文件的 Web 应用程序)；
    <template>
  <div>
    <el-button
      type="text"
      class="download-btn"
      @click="getFormFile(file)"
      >打印PDF文件</el-button
    >
    <el-button
      type="text"
      class="download-btn"
      @click="downloadPdf"
      >下载PDF文件</el-button
    >
  </div>
</template>

<script>
import FileSaver from 'file-saver';

// 获取PDF文档流接口接口url
export const getFormFile = params =>
  http.get('/api/xxx/Form/GetFormFile', {
    responseType: 'blob',
    params
  });

export default {
  methods: {
    // 1. 打印PDF文件
    async printPdf(file) {
      const params = {
        Id: file.id,
        FormId: this.formData?.id
      };
      // 调接口获取文档流
      const res = await getFormFile(params);
      // 如果返回的不是pdf则抛出错误提示
      if (res.type === 'application/json') {
        this.$message.error('找不到对应的文件');
        return;
      }
      // 将文档流转为blob地址，放到js创建的隐藏iframe标签内；再执行print()方法。
      if (res.type === 'application/pdf') {
        const blob = new Blob([res], { type: 'application/pdf' });
        var date = new Date().getTime();
        var ifr = document.createElement('iframe');
        ifr.style.frameborder = 'no';
        ifr.style.display = 'none';
        ifr.style.pageBreakBefore = 'always';
        ifr.setAttribute('id', 'printPdf' + date);
        ifr.setAttribute('name', 'printPdf' + date);
        ifr.src = window.URL.createObjectURL(blob);
        document.body.appendChild(ifr);
        this.doPrint('printPdf' + date);
        window.URL.revokeObjectURL(ifr.src); // 释放URL 对象
      }
    },
    doPrint(val) {
      // 使用`contentWindow`属性来访问iframe中的文档对象
      var ordonnance = document.getElementById(val).contentWindow;
      setTimeout(() => {
        ordonnance.print();
      }, 100);
    }
  },

  // 2. 下载PDF文件
  downloadPdf() {
    // 安装 npm install file-saver --save

    FileSaver.saveAs(blob, file.fileName);
    // FileSaver saveAs(Blob/File/Url,fileType)
    // 参数1：支持的类型：blob（二进制）、File（文件）、Url（url链接）
    // 参数3: 文件名称
    // 参数2：文件类型
  }
};
</script>

    # 二、全屏水印

    1. wm.js — 封装水印函数
    function getImageDataUrl(msg, config) {
  config = config || {};
  let fontSize = 30;
  let canvas = document.createElement('canvas');
  canvas.width = config.width || fontSize * msg.length * 0.7;
  canvas.height = config.height || fontSize * msg.length * 0.4;
  let ctx = canvas.getContext('2d');
  ctx.translate(fontSize / 3, canvas.height - fontSize / 3);
  ctx.rotate(((config.rotate || -30) * Math.PI) / 180);
  ctx.font = fontSize + 'px Arial';
  ctx.fillStyle = config.color || '#eee';
  ctx.fillText(msg, 0, 0);
  return canvas.toDataURL('png');
}
export default function setWatermark(el, msg, config) {
  let content = msg;
  if (typeof msg === 'function') {
    content = msg();
    if (!content) {
      return setTimeout(() => {
        setWatermark(el, msg, config);
      }, 10);
    }
  }
  setTimeout(() => {
    let img = getImageDataUrl(content, config);
    el.style.background = `repeat url(${img})`;
  }, 0);
}

    # 三、require.context()

    require.context参数介绍：
    require.context(directory,useSubdirectories,regExp)

    • directory:表示检索的目录
    • useSubdirectories：表示是否检索子文件夹
    • regExp:匹配文件的正则表达式,一般是文件名
    1. 在main.js中引入大量公共组件
      下面的代码都是在main.js，也就是项目的入口文件中写的，在这个地方将组件进行一次性的全局注册，这么多的代码写在main.js中可能显得比较累赘，我们也可以使用方法2（Vue插件），使得代码看起来比较简介一些。
    import Vue from 'vue'
// 自定义组件
const requireComponents = require.context('../views/components', true, /\.vue/)
// 打印结果
// 遍历出每个组件的路径
requireComponents.keys().forEach(fileName => {
  // 组件实例
  const reqCom = requireComponents(fileName)
  // 截取路径作为组件名
  const reqComName =reqCom.name|| fileName.replace(/\.\/(.*)\.vue/,'$1')
  // 组件挂载
  Vue.component(reqComName, reqCom.default || reqCom)
})

    1. 引入vuex的module
      index.js代码(modules目录下的)
    // src/store/index.js
import Vue from "vue";
import Vuex from "vuex";
import state from "./state";
import getters from "./getters";
import mutations from "./mutations";
import actions from "./actions";

Vue.use(Vuex);

// 采用require.context自动引入modules中的模块文件
const modulesFiles = require.context("./modules", true, /\.js$/);

// 不需要使用 `import permission from "./modules/permission"`来引入
// 定义在modules目录中，则自动引入模块文件
const modules = modulesFiles.keys().reduce((modules, modulePath) => {
  // set './app.js' => 'app'
  const moduleName = modulePath.replace(/^\.\/(.*)\.\w+$/, "$1");
  const value = modulesFiles(modulePath);
  modules[moduleName] = value.default;
  return modules;
}, {});

export default new Vuex.Store({
  state,
  getters,
  mutations,
  actions,
  modules
});
